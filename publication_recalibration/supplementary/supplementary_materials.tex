\section{S1: Classification Method Details}

\subsection{S1.1 Zero-Shot BART-MNLI}

\begin{equation}
\hat{y} = \arg\max_{c \in \mathcal{C}} P(\text{entailment} \mid \text{description}, c)
\end{equation}
where $\mathcal{C} = \{\text{Contraindicated, Major, Moderate, Minor}\}$.

\subsection{S1.2 Confidence-Weighted Adjustment}

\begin{equation}
\text{Severity}_{\text{adj}} = 
\begin{cases}
\text{downgrade by 1 level} & \text{if } \hat{y} \in \{\text{Contraindicated, Major}\} \land c < 0.65 \\
\hat{y} & \text{otherwise}
\end{cases}
\end{equation}
where $\hat{y}$ is the predicted severity and $c$ is the model's confidence score.

\subsection{S1.3 Rule-Based Classifier}

\subsubsection{S1.3.1 Weight Derivation Equation}

\begin{equation}
w_k = \ln\left(\frac{P(k|\text{Major})}{P(k|\text{Moderate})}\right)
\end{equation}

LR = Likelihood Ratio = $P(k|\text{Major}) / P(k|\text{Moderate})$. Weight = $\ln(\text{LR})$.

\subsubsection{S1.3.3 Algorithm 1: Keyword Weight Derivation}

\textit{Input:} $\mathcal{D}_{\text{train}}$ (training set), $\mathcal{K}$ (candidate keywords)

\textit{Output:} $\mathcal{W}$ (keyword $\rightarrow$ weight dictionary)

\begin{enumerate}
\item Count Major and Moderate instances: $n_{\text{Major}}$, $n_{\text{Moderate}}$
\item For each keyword $k \in \mathcal{K}$:
\begin{itemize}
\item Count occurrences: $c_{\text{Major}}$, $c_{\text{Moderate}}$
\item If $c_{\text{Major}} + c_{\text{Moderate}} \geq 50$:
\begin{itemize}
\item $P(k|\text{Major}) \gets c_{\text{Major}} / n_{\text{Major}}$
\item $P(k|\text{Moderate}) \gets c_{\text{Moderate}} / n_{\text{Moderate}}$
\item $\mathcal{W}[k] \gets \ln(P(k|\text{Major}) / P(k|\text{Moderate}))$
\end{itemize}
\end{itemize}
\item Return $\mathcal{W}$
\end{enumerate}

\subsubsection{S1.3.4 Algorithm 2: Severity Classification}

\textit{Input:} $\mathcal{W}$ (keyword weights), $\mathcal{S}_{\text{train}}$ (training scores), $d$ (DDI description)

\textit{Output:} severity $\in$ \{Contraindicated, Major, Moderate, Minor\}

\textbf{Step 1: Compute severity score}
\begin{equation}
S = \sum_{k \in \mathcal{W}} \mathcal{W}[k] \cdot \mathbb{1}[k \in d]
\end{equation}

\textbf{Step 2: Compute percentile thresholds from training scores (grid-search optimized)}
\begin{itemize}
\item $P_{96} \gets \text{percentile}(\mathcal{S}_{\text{train}}, 96)$ (top 4\%)
\item $P_{92} \gets \text{percentile}(\mathcal{S}_{\text{train}}, 92)$ (top 8\%, optimized from 78\%)
\item $P_{2} \gets \text{percentile}(\mathcal{S}_{\text{train}}, 2)$ (bottom 2\%, optimized from 4\%)
\end{itemize}

\textbf{Step 3: Classify based on score}
\begin{equation}
\text{severity} = 
\begin{cases}
\text{Contraindicated} & \text{if } S \geq P_{96} \\
\text{Major} & \text{if } P_{92} \leq S < P_{96} \\
\text{Minor} & \text{if } S \leq P_{2} \\
\text{Moderate} & \text{otherwise}
\end{cases}
\end{equation}

\section{S2: Evidence-Based Classifier}

Uses identical keyword weights as Rule-Based (Table~\ref{tab:empirical_weights}) with an additional confidence penalty:

\begin{equation}
S_{\text{adjusted}} = 
\begin{cases}
S_{\text{raw}} \times 0.8 & \text{if confidence} < 0.5 \\
S_{\text{raw}} & \text{otherwise}
\end{cases}
\end{equation}

\textbf{Note:} Achieves identical performance (66.4\% exact accuracy) to Rule-Based because confidence penalization has minimal effect on the empirically-weighted scores.

\section{S3: Statistical Analysis Details}

\subsection{S3.1 McNemar's Test for Paired Classifier Comparison}

McNemar's test compares two classifiers on the same test set by constructing a 2$\times$2 contingency table:

\begin{table}[h]
\centering
\caption{McNemar's Contingency Table (Rule-Based vs.~Zero-Shot)}
\begin{tabular}{@{}lcc@{}}
\toprule
 & \textbf{Zero-Shot Correct} & \textbf{Zero-Shot Wrong} \\
\midrule
\textbf{Rule-Based Correct} & 502 & 6,900 \\
\textbf{Rule-Based Wrong} & 1,962 & 1,786 \\
\bottomrule
\end{tabular}
\end{table}

\begin{equation}
\chi^2 = \frac{(|b - c| - 1)^2}{b + c} = \frac{(|1962 - 6900| - 1)^2}{8862} = 2750.4
\end{equation}

where $b$ = Zero-Shot only correct (1,962) and $c$ = Rule-Based only correct (6,900). Result: $\chi^2 = 2750.4$, df=1, $p < 2.2 \times 10^{-16}$.

\subsection{S3.2 Wilson Score Confidence Intervals}

For accuracy proportions, we use the Wilson score interval (more accurate than normal approximation for bounded proportions):

\begin{equation}
\text{CI} = \frac{\hat{p} + \frac{z^2}{2n} \pm z\sqrt{\frac{\hat{p}(1-\hat{p})}{n} + \frac{z^2}{4n^2}}}{1 + \frac{z^2}{n}}
\end{equation}

where $\hat{p}$ is observed accuracy, $n$ = 11,150, and $z$ = 1.96 for 95\% CI.

\textbf{Results:}
\begin{itemize}
\item Zero-Shot: 22.1\% (95\% CI: 21.3\%--22.9\%)
\item Rule-Based: 66.4\% (95\% CI: 65.5\%--67.3\%)
\end{itemize}

\subsection{S3.3 Jensen-Shannon Divergence}

\begin{equation}
D_{JS}(P \| Q) = \frac{1}{2} D_{KL}(P \| M) + \frac{1}{2} D_{KL}(Q \| M)
\end{equation}
where $M = \frac{1}{2}(P + Q)$.

\subsection{S3.4 Chi-Square Goodness of Fit}

\begin{equation}
\chi^2 = \sum_{i} \frac{(O_i - E_i)^2}{E_i}
\end{equation}

\subsection{S3.5 Cohen's Kappa Significance Test}

To test whether $\kappa$ is significantly different from zero:

\begin{equation}
z = \frac{\kappa}{\text{SE}_{\kappa}}, \quad \text{where } \text{SE}_{\kappa} \approx \sqrt{\frac{p_e}{n(1-p_e)^2}}
\end{equation}

For our Rule-Based classifier: $\kappa = 0.096$, SE $\approx$ 0.020, $z = 4.77$, $p < 0.0001$.

The improvement from baseline ($\kappa = 0.000$) demonstrates meaningful classifier calibration.
